df_prez = pd.read_csv('presidents.csv')
state = df_prez['State']
state = dict(state.value_counts())

x = [i for i in state.keys()]
freq = [state[i] for i in state.keys()]

fig,ax=plt.subplots()
ax.bar(x=x, height = freq, color = 'xkcd:ultramarine', alpha = .75)
ax.set_ylim(0,max(freq)+.5)

fig.suptitle('President Birthplace Frequency Histogram')
fig.set_size_inches(16,5)

ax.xaxis.set_label_text('State of Birth')
ax.yaxis.set_label_text('Frequency')

plt.show()
